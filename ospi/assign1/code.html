let gamma' n =
  let m = ones n n in
    for row = 0 to (n-1) do
      for col = 1 to (n/2) do
        let value = ((float (row+1)) *. (float col) *. 2. *. pi) /. (float n) in
          (m => (row, col*2-1)) (cos value);
          if (col*2) < n then (m => (row,col*2)) (sin value);
      done
    done;
    m

let gamma n =
  let m = gamma' n in
  let f1 = 1. /. (float n) in
  let f2 = 2. /. (float n) in
  (*let f = 1. in*)
    M.transpose_in_place m;
    for col = 0 to (n-1) do
      (m => (0,col)) ((m $@ (0,col)) *. f1)
    done;
    for row = 1 to (n-1) do
      for col = 0 to (n-1) do
        (m => (row,col)) ((m $@ (row,col)) *. f2)
      done
    done;
    m
